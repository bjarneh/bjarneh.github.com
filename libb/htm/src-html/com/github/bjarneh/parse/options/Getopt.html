016    package com.github.bjarneh.parse.options;
017    
018    import java.util.HashMap;
019    import java.util.ArrayList;
020    
021    /**
022     *
023     * Getopt is a command line parser.
024     *
025     * <b>Not</b> a re-implementation of GNU-getopt.
026     
027    <pre>
028     
029       // Typical use:
030    
031       Getopt getopt = new Getopt();
032    
033       getopt.addBoolOption("-h -help --help help");
034       getopt.addBoolOption"("-v -version --version");
035    
036       // addFancyStrOption creates all variants of a flag, i.e.
037       // these two statements are equivalent
038       //
039       // 1. getopt.addStrOption("-p -p= -port -port= --port --port=");
040       // 
041       // 2. getopt.addFancyStrOption("-p --port");
042       
043       getopt.addFancyStrOption("-p --port");
044       getopt.addFancyStrOption("-r --root");
045       getopt.addStrOption("-I -I=");
046    
047       String[] rest = getopt.parse(args);
048    
049       if(getopt.isSet("-help")){
050         System.out.println(" I will help you ");
051       }
052    
053       if(getopt.isSet("-port")){
054         int port = getopt.getInt("-port");
055       }
056    
057       if(getopt.isSet("-root")){
058         String root = getopt.get("-root");
059       }
060    
061       if(getopt.isSet("-I")){
062         for(String i: getopt.getAll("-I")){
063           System.out.printf(" -I: %s\n", i);
064         }
065       }
066       
067       for(String s: rest){
068         System.out.printf("unparsed: %s\n", s);
069       }
070    
071    
072     </pre>
073     *
074     *
075     * @version 1.0
076     * @author bjarneh@ifi.uio.no
077     */
078    
079    public class Getopt {
080    
081        ArrayList<Option> options;
082        HashMap<String, Option> cache;
083        private boolean dieOnError;
084    
085        /**
086         * Constructor alias for Getopt(true), ie die on error.
087         */
088        public Getopt(){
089            this(true);// die on error
090        }
091    
092        /**
093         * Constructor for Getopt which allows us to be firm or hysteric.
094         * @param dieOnError die on parse error or throw exception.
095         */
096        public Getopt(boolean dieOnError){
097            options = new ArrayList<Option>(50);
098            cache = new HashMap<String, Option>();
099            dieOnError = dieOnError;
100        }
101    
102        /**
103         * Add a boolean flag.
104         * @param flagStr white space separated flags
105         */
106        public void addBoolOption(String flagStr){
107    
108            String[] flags = splitFlags(flagStr);
109            Option opt   = new BoolOption(flags);
110            options.add(opt);
111    
112            for(String f : flags){
113                cache.put(f, opt);
114            }
115        }
116    
117        /**
118         * Add a flag that expects options.
119         * @param flagStr white space separated flags
120         */
121        public void addStrOption(String flagStr){
122    
123            String[] flags = splitFlags(flagStr);
124            Option opt   = new StrOption(flags);
125            options.add(opt);
126    
127            for(String f : flags){
128                cache.put(f, opt);
129            }
130        }
131    
132    
133        /**
134         * Add a flag that expects options, this method
135         * automatically generates some flags for you.
136         *
137         * <pre>
138         * addFancyStrOption("-f --file");
139         * // is equivalent to
140         * addStrOption("-f -f= -file -file= --file --file=");
141         * </pre>
142         *
143         * @param flagStr white space separated flags
144         */
145        public void addFancyStrOption(String flagStr){
146    
147            String[] flags = makeFancyOptions(flagStr);
148            Option opt   = new StrOption(flags);
149            options.add(opt);
150    
151            for(String f : flags){
152                cache.put(f, opt);
153            }
154        }
155    
156        /**
157         * Parse arguments with options given.
158         *
159         * @param argv input arguments
160         * @return unparsed arguments
161         */
162        public String[] parse(String[] argv){
163    
164            Option opt;
165            StrOption stropt;
166            BoolOption boolopt;
167            ArrayList<String> rest = new ArrayList<String>();
168    
169            for(int i = 0; i < argv.length; i++){
170    
171                if(cache.containsKey(argv[i])){
172    
173                    opt = cache.get(argv[i]);
174    
175                    if(opt instanceof BoolOption){
176    
177                        boolopt = (BoolOption) opt;
178                        boolopt.found();
179    
180                    }else if(opt instanceof StrOption) {
181    
182                        if( i+1 >= argv.length){
183                            String errmsg = "Missing argument for: %s" ;
184                            errmsg = String.format(errmsg, argv[i]);
185                            if(! dieOnError){
186                                throw new Error(errmsg);
187                            }else{
188                                System.err.printf("%s\n", errmsg);
189                                System.exit(1);
190                            }
191                        }
192    
193                        stropt = (StrOption) opt;
194                        stropt.add(argv[++i]);
195                    }
196    
197                }else{ // not plain options 
198    
199                    if( ! juxtaBool( argv[i] ) && ! juxtaStr(argv[i]) ){
200                        rest.add( argv[i] );
201                    }
202                }
203            }
204    
205            String[] tmp = new String[rest.size()];
206            return rest.toArray(tmp);
207        }
208    
209        // -xvzf == -x -v -z -f
210        private boolean juxtaBool(String flag){
211    
212            if( flag.length() < 3 ){ return false; }
213    
214            String tmp;
215            Option opt;
216            BoolOption boolopt;
217    
218            char[] c = flag.toCharArray();
219    
220            for(int i = 1; i < c.length; i++){
221                tmp = "-" + c[i];
222                opt = cache.get(tmp);
223                if( opt == null || ! (opt instanceof BoolOption) ){
224                    return false;
225                }
226            }
227    
228            for(int i = 1; i < c.length; i++){
229                tmp = "-" + c[i];
230                boolopt = (BoolOption) cache.get(tmp);
231                boolopt.found();
232            }
233    
234            return true;
235        }
236    
237        // -I/dir == -I /dir
238        private boolean juxtaStr(String flag){
239    
240            if( ! flag.startsWith("-") ){ return false; }
241    
242            int maxLen = -1;
243            String maxFlag = "";
244            Option opt;
245            StrOption stropt;
246    
247            for(String s : cache.keySet()){
248                if( flag.startsWith(s) ){
249                    opt = cache.get(s);
250                    if( opt instanceof StrOption ){
251                        if( s.length() > maxLen ){
252                            maxFlag = s;
253                            maxLen  = s.length();
254                        }
255                    }
256                }
257            }
258    
259            if( maxLen > 0 ){
260                stropt = (StrOption) cache.get(maxFlag);
261                stropt.add( flag.substring(maxFlag.length()) );
262                return true;
263            }
264    
265            return false;
266        }
267    
268        private String[] makeFancyOptions(String flagStr){
269    
270            ArrayList<String> fancy = new ArrayList<String>();
271            String[] flags = splitFlags(flagStr);
272    
273            for(String f: flags){
274                fancy.add(f);
275                if(f.startsWith("--")){
276                    fancy.add(f.substring(1));
277                    fancy.add(f.substring(1) + "=");
278                    fancy.add(f + "=");
279                }else if(f.startsWith("-")){
280                    fancy.add(f + "=");
281                }
282            }
283    
284            String[] fancyFlags = new String[fancy.size()];
285            return fancy.toArray(fancyFlags);
286        }
287    
288        /**
289         * Report if a flag was set during {@link Getopt#parse}.
290         * @param flag that belongs to one of the options
291         * @return true if flag was encountered during parse
292         */
293        public boolean isSet(String flag){
294            Option opt = cache.get(flag);
295            return opt.isSet();
296        }
297    
298        /**
299         * Return the argument of a string option.
300         *
301         * @param flag a flag belonging to a string option
302         * @return argument for option denoted by flag
303         */
304        public String get(String flag){
305            StrOption stropt = (StrOption) cache.get(flag);
306            return stropt.get();
307        }
308    
309        /**
310         * Return all argument for a string option.
311         *
312         * To give multiple arguments to a string option
313         * you need to give the flag multiple times, i.e.
314         *
315         * <pre>
316         * -I/usr/lib -I/usr/share/lib   // OK
317         * -I /usr/lib /usr/share/lib    // NOT OK
318         * </pre>
319         *
320         * @param flag a flag belonging to a string option
321         * @return an array of arguments belonging to a string option
322         */
323        public String[] getAll(String flag){
324            StrOption stropt = (StrOption) cache.get(flag);
325            return stropt.getAll();
326        }
327    
328        /**
329         * Return argument for a flag as an int.
330         *
331         * @param flag a flag belonging to a string option
332         * @return the argument of the flag converted to an int
333         */
334        public int getInt(String flag){
335            return Integer.parseInt( get(flag) );
336        }
337    
338        /**
339         * Return argument for a flag as a float.
340         *
341         * @param flag a flag belonging to a string option
342         * @return the argument of the flag converted to a float
343         */
344        public float getFloat(String flag){
345            return Float.parseFloat( get(flag) );
346        }
347    
348        /**
349         * Return argument for a flag as a double.
350         *
351         * @param flag a flag belonging to a string option
352         * @return the argument of the flag converted to a double
353         */
354        public double getDouble(String flag){
355            return Double.parseDouble( get(flag) );
356        }
357    
358        private String[] splitFlags(String flagStr){
359            return flagStr.trim().split("\\s+");
360        }
361    
362        /**
363         * Reset all options to their original state (false/empty).
364         */
365        public void reset(){
366            for(Option opt: options){
367                opt.reset();
368            }
369        }
370    
371        /**
372         * Returns a string a short description of options and their state.
373         * @return string representation of Getopt, i.e. flags and values
374         */
375        public String toString(){
376            StringBuilder sb = new StringBuilder();
377            for(Option opt : options){
378                sb.append(opt).append("\n");
379            }
380            return sb.toString();
381        }
