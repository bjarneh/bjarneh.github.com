033    /**
034     * Common io functions.
035     *
036     * <pre>
037     *
038     *  // Typical use:
039     *
040     *  byte[] bytes;
041     *  
042     *  // fetch file as raw byte array
043     *  bytes = io.raw("somefile.txt");
044     *
045     *  // fetch URL as raw byte array
046     *  bytes = io.wget("http://example.com/index.html");
047     *
048     *  // fetch stdin as raw byte array
049     *  bytes = io.raw(System.in);
050     *
051     *  // pipe an input stream to an output stream (stdin to stdout)
052     *  io.pipe(System.in, System.out);
053     *
054     *  // file slurp
055     *  String content = new String(io.raw("somefile.txt"));
056     *
057     *  // file slurp with specified encoding
058     *  String content = new String(io.raw(System.in), "ISO-8859-1");
059     *
060     * </pre>
061     * <b>note</b>: There is a lot of comments here since this is
062     * meant to be a library with some documentation, i.e. make 
063     * sure you turn on your comment folding before you start 
064     * reading this file.
065     *
066     * @version 1.0
067     * @author  bjarneh@ifi.uio.no
068     */

082        /**
083         * In memory raw file slurp.
084         *
085         * It should be noted that I'm just guessing that this is the
086         * best way to do things based on votes from stackoverflow,
087         * which could very well be wrong; but if it is I'm not alone :-)
088         *
089         * stackoverflow: 326390
090         *
091         * @param stream to slurp and return as raw bytes
092         * @return a byte array containing the bytes of the stream
093         */
094        public static byte[] raw(FileInputStream stream) throws IOException {
095            try {
096                FileChannel fc = stream.getChannel();
097                MappedByteBuffer bb =
098                    fc.map(FileChannel.MapMode.READ_ONLY, 0, fc.size());
099                if(bb.hasArray()){
100                    return bb.array();
101                }else{
102                    byte[] b = new byte[(int)fc.size()];
103                    bb.get(b);
104                    return b;
105                }
106            }
107            finally {
108                stream.close();
109            }
110        }

112        /**
113         * 
114         * Alias for io.raw(new File(fname)).

116         * @param fname file-name to slurp and return as raw bytes

120            return raw(new File(fname));

123        /**
124         * Alias for io.raw(new FileInputStream(file)).
125         *
126         * @param file file object to slurp and return as raw bytes
127         * @return a byte array containing the bytes of the file 
128         */
129        public static byte[] raw(File file) throws IOException {
130            return raw(new FileInputStream(file));
131        }
132    
133        /**
134         * Use io.pipe to write to an ByteArrayOutputStream, and fetch array.
135         *
136         * @param input the stream we want to return as raw bytes
137         * @return a byte array containing the bytes of the input stream
138         */
139        public static byte[] raw(InputStream input) throws IOException {
140            ByteArrayOutputStream output = new ByteArrayOutputStream();
141            pipe(input, output);
142            return output.toByteArray();
143        }

149        /**
150         * Write bytes from an input stream to an output stream.
151         *
152         * @param fi  where bytes are read
153         * @param fo  where bytes are written
154         * @param bufferSize is the size of the buffer used
155         */
156        public static void pipe(InputStream fi, OutputStream fo, int bufferSize)
157            throws IOException
158        {
159            int got = -1;
160            byte[] b = new byte[bufferSize];
161    
162            for( got = fi.read(b); got > 0; got = fi.read(b) ){
163                fo.write(b, 0, got);
164            }
165        }
166    
167        /**
168         * Alias for io.pipe(fi, fo, io.DEFAULT_BUFFER_SIZE).
169         *
170         * @param fi  where bytes are read
171         * @param fo  where bytes are written
172         */
173        public static void pipe(InputStream fi, OutputStream fo)
174            throws IOException
175        {
176            pipe(fi, fo, DEFAULT_BUFFER_SIZE);
177        }
178    
179        /**
180         * Alias for io.pipe(new ByteArrayInputStream(b), fo, io.DEFAULT_BUFFER_SIZE).
181         *
182         * @param b  the bytes
183         * @param fo  where bytes are written
184         */
185        public static void pipe(byte[] b, OutputStream fo)
186            throws IOException
187        {
188            pipe(new ByteArrayInputStream(b), fo, DEFAULT_BUFFER_SIZE);
189        }

237        /**
238         * Unzip files without the hassle.
239         * @param file to be unzipped
240         */
241        public static void unzip( File file ) throws Exception {
242    
243            ZipFile zipFile = new ZipFile( file );
244    
245            File parentFile, tmpFile, grandFather;
246            InputStream stream;
247            FileOutputStream output;
248    
249            String parent = file.getParent();
250    
251            if( parent != null ){
252                parentFile = new File(parent);
253            }else{
254                parentFile = new File(".");
255            }
256    
257            for(ZipEntry e : handy.toList( zipFile.entries() )){
258                if( e.isDirectory() ){
259                    new File(parentFile, e.getName()).mkdirs();
260                }else{
261                    stream      = zipFile.getInputStream(e);
262                    tmpFile     = new File(parentFile, e.getName());
263                    grandFather = tmpFile.getParentFile();
264                    if( grandFather != null ){
265                        if( !grandFather.exists() || !grandFather.isDirectory() ){
266                            grandFather.mkdirs();
267                        }
268                    }
269                    output      = new FileOutputStream( tmpFile );
270                    pipe( stream, output );
271                }
272            }
273        }
