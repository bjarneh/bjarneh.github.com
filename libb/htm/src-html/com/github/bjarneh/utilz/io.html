016    package com.github.bjarneh.utilz;
017    
018    // stdlib
019    import java.net.URL;
020    import java.io.File;
021    import java.io.InputStream;
022    import java.io.OutputStream;
023    import java.io.FileInputStream;
024    import java.io.IOException;
025    import java.io.ByteArrayInputStream;
026    import java.io.ByteArrayOutputStream;
027    import java.nio.MappedByteBuffer;
028    import java.nio.channels.FileChannel;
029    
030    /**
031     * Common io functions.
032     *
033     * <pre>
034     *
035     *  // Typical use:
036     *
037     *  byte[] bytes;
038     *  
039     *  // fetch file as raw byte array
040     *  bytes = io.raw("somefile.txt");
041     *
042     *  // fetch URL as raw byte array
043     *  bytes = io.wget("http://example.com/index.html");
044     *
045     *  // fetch stdin as raw byte array
046     *  bytes = io.raw(System.in);
047     *
048     *  // pipe an input stream to an output stream (stdin to stdout)
049     *  io.pipe(System.in, System.out);
050     *
051     *  // file slurp
052     *  String content = new String(io.raw("somefile.txt"));
053     *
054     *  // file slurp with specified encoding
055     *  String content = new String(io.raw(System.in), "ISO-8859-1");
056     *
057     * </pre>
058     * <b>note</b>: There is a lot of comments here since this is
059     * meant to be a library with some documentation, i.e. make 
060     * sure you turn on your comment folding before you start 
061     * reading this file.
062     *
063     * @version 1.0
064     * @author  bjarneh@ifi.uio.no
065     */
066    
067    public class io {
068    
069        // we only have functions
070        private io(){}
071    
072        public final static int DEFAULT_BUFFER_SIZE = 4096;
073    
074    
075        ////////////////////
076        // raw file slurp //
077        ////////////////////
078    
079        /**
080         * In memory raw file slurp.
081         *
082         * It should be noted that I'm just guessing that this is the
083         * best way to do things based on votes from stackoverflow,
084         * which could very well be wrong; but if it is I'm not alone :-)
085         *
086         * stackoverflow: 326390
087         *
088         * @param stream to slurp and return as raw bytes
089         * @return a byte array containing the bytes of the stream
090         */
091        public static byte[] raw(FileInputStream stream) throws IOException {
092            try {
093                FileChannel fc = stream.getChannel();
094                MappedByteBuffer bb =
095                    fc.map(FileChannel.MapMode.READ_ONLY, 0, fc.size());
096                if(bb.hasArray()){
097                    return bb.array();
098                }else{
099                    byte[] b = new byte[(int)fc.size()];
100                    bb.get(b);
101                    return b;
102                }
103            }
104            finally {
105                stream.close();
106            }
107        }
108    
109        /**
110         * 
111         * Alias for io.raw(new File(fname)).
112         *
113         * @param fname file-name to slurp and return as raw bytes
114         * @return a byte array containing the bytes of the file 
115         */
116        public static byte[] raw(String fname) throws IOException {
117            return raw(new File(fname));
118        }
119    
120        /**
121         * Alias for io.raw(new FileInputStream(file)).
122         *
123         * @param file file object to slurp and return as raw bytes
124         * @return a byte array containing the bytes of the file 
125         */
126        public static byte[] raw(File file) throws IOException {
127            return raw(new FileInputStream(file));
128        }
129    
130        /**
131         * Use io.pipe to write to an ByteArrayOutputStream, and fetch array.
132         *
133         * @param input the stream we want to return as raw bytes
134         * @return a byte array containing the bytes of the input stream
135         */
136        public static byte[] raw(InputStream input) throws IOException {
137            ByteArrayOutputStream output = new ByteArrayOutputStream();
138            pipe(input, output);
139            return output.toByteArray();
140        }
141    
142        ////////////////////////////////////////////
143        // piping input streams to output streams //
144        ////////////////////////////////////////////
145    
146        /**
147         * Write bytes from an input stream to an output stream.
148         *
149         * @param fi  where bytes are read
150         * @param fo  where bytes are written
151         * @param bufferSize is the size of the buffer used
152         */
153        public static void pipe(InputStream fi, OutputStream fo, int bufferSize)
154            throws IOException
155        {
156            int got = -1;
157            byte[] b = new byte[bufferSize];
158    
159            for( got = fi.read(b); got > 0; got = fi.read(b) ){
160                fo.write(b, 0, got);
161            }
162        }
163    
164        /**
165         * Alias for io.pipe(fi, fo, io.DEFAULT_BUFFER_SIZE).
166         *
167         * @param fi  where bytes are read
168         * @param fo  where bytes are written
169         */
170        public static void pipe(InputStream fi, OutputStream fo)
171            throws IOException
172        {
173            pipe(fi, fo, DEFAULT_BUFFER_SIZE);
174        }
175    
176        /**
177         * Alias for io.pipe(new ByteArrayInputStream(b), fo, io.DEFAULT_BUFFER_SIZE).
178         *
179         * @param b  the bytes
180         * @param fo  where bytes are written
181         */
182        public static void pipe(byte[] b, OutputStream fo)
183            throws IOException
184        {
185            pipe(new ByteArrayInputStream(b), fo, DEFAULT_BUFFER_SIZE);
186        }
187    
188    
189        /**
190         * Fetch the bytes of a String representing an URL.
191         * @param url represented as a String
192         * @return an array of bytes fetched from URL
193         */
194        public static byte[] wget(String url)
195            throws IOException
196        {
197            return wget(new URL(url));
198        }
199    
200        /**
201         * Fetch the bytes of an URL.
202         * @param url to read
203         * @return an array of bytes fetched from URL
204         */
205        public static byte[] wget(URL url)
206            throws IOException
207        {
208            return raw(url.openStream());
209        }
210    
211        /**
212         * Write the bytes of an URL to an OutputStream.
213         * @param url to read from
214         * @param fo output stream to write bytes to
215         * @param bufferSize size of buffer used by io.pipe
216         */
217        public static void wget(URL url, OutputStream fo, int bufferSize)
218            throws IOException
219        {
220            pipe(url.openStream(), fo, bufferSize);
221        }
222    
223        /**
224         * Alias for wget(url, fo, io.DEFAULT_BUFFER_SIZE).
225         * @param url to read from
226         * @param fo output stream to write bytes to
227         */
228        public static void wget(URL url, OutputStream fo)
229            throws IOException
230        {
231            pipe(url.openStream(), fo, DEFAULT_BUFFER_SIZE);
232        }
233    }
