026    /**
027     * Collecting some handy functions that can be useful all over the place.
028     *
029     * <pre>
030     *
031     * // Typical use:
032     *
033     * String[] lines = getLines(); // assume this does something
034     * String content = handy.join("\n", lines);
035     *
036     * ArrayList&lt;String&gt; list = new ArrayList&lt;String&gt;();
037     * handy.add( list, lines );
038     *
039     * String s = handy.dynReplace("&lt;b&gt;([^&lt;]+)&lt;\\/b&gt;", "&lt;b&gt;b1&lt;/b&gt; txt &lt;b&gt;b2&lt;/b&gt;");
040     * s.equals("b1 txt b2"); // true
041     *
042     * </pre>
043     *
044     * <b>note</b>: There is a lot of comments here since this is
045     * meant to be a library with some documentation, i.e. make 
046     * sure you turn on your comment folding before you start 
047     * reading this file.
048     *
049     * @version 1.0
050     * @author  bjarneh@ifi.uio.no
051     */
052    
053    public class handy {
054    
055        // we only have functions
056        private handy(){}
057    
058        /**
059         * Join elements of Iterable&lt;? extends Object&gt; with
060         * separator using {@link Object#toString}.
061         *
062         * @param sep the separator to use
063         * @param elements to join into a String
064         * @return all the elements joined by 'sep'
065         */
066        public static String joinAny(String sep,
067                                     Iterable<? extends Object> elements){
068    
069            StringBuilder sb = new StringBuilder("");
070            Iterator<? extends Object> it = elements.iterator();
071    
072            while( it.hasNext() ){
073                sb.append(it.next().toString());
074                if( it.hasNext() ){ sb.append(sep); }
075            }
076    
077            return sb.toString();
078        }
079    
080        /**
081         * Join elements of Iterable&lt;String&gt; with separator to a String.
082         *
083         * @param sep the separator to use
084         * @param elements to join into a String
085         * @return all the elements joined by 'sep'
086         */
087        public static String join(String sep, Iterable<String> elements){
088    
089            StringBuilder sb = new StringBuilder("");
090            Iterator<String> it = elements.iterator();
091    
092            while( it.hasNext() ){
093                sb.append(it.next());
094                if( it.hasNext() ){ sb.append(sep); }
095            }
096    
097            return sb.toString();
098        }
099    
100        /**
101         * Join elements of String array with separator to a String.
102         *
103         * @param sep the separator to use
104         * @param elements to join into a String
105         * @return all the elements joined by 'sep'
106         */
107        public static String join(String sep, String[] elements){
108    
109            StringBuilder sb = new StringBuilder("");
110    
111            for(int i = 0; i < elements.length; i++){
112                sb.append(elements[i]);
113                if(i+1 < elements.length){
114                    sb.append(sep);
115                }
116            }
117            return sb.toString();
118        }
119    
120        /**
121         * Add array elements to a Collection of same type.
122         * @param collection that we would like to add array elements to
123         * @param array that we would like to add to collection
124         */
125        public static <T> void add(Collection<T> collection, T[] array){
126            for(T t: array){
127                collection.add(t);
128            }
129        }
130    
131        /**
132         * Replace every occurence of regexp (containing match-group) with
133         * match-group (sort of like $1 in some regexp implementations).
134         *
135         * @param regexpWithMatchGroup a pattern containing a match-group
136         * @param str the string we want to replace stuff in
137         * @return a string were pattern is replaced with match-group
138         */
139        public static String dynReplace(String regexpWithMatchGroup, String str){
140            return dynReplace( Pattern.compile( regexpWithMatchGroup ), str);
141        }
142    
143        /**
144         * Replace every occurence of pattern (containing match-group) with
145         * match-group (sort of like $1 in some regexp implementations).
146         *
147         * @param patternWithMatchGroup a pattern containing a match-group
148         * @param str the string we want to replace stuff in
149         * @return a string were pattern is replaced with match-group
150         */
151        public static String dynReplace(Pattern patternWithMatchGroup, String str){
152    
153            String tmp = str;
154            Matcher m  = patternWithMatchGroup.matcher(str);
155    
156            while(m.find()){
157                tmp = m.replaceFirst( m.group(1) );
158                m = patternWithMatchGroup.matcher( tmp );
159            }
160    
161            return tmp;
162        }
163    }
