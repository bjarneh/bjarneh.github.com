034     * Collecting some handy functions that can be useful all over the place.
035     *
036     * <pre>
037     *
038     * // Typical use:
039     *
040     * String[] lines = getLines(); // assume this does something
041     * String content = handy.join("\n", lines);
042     *
043     * ArrayList&lt;String&gt; list = new ArrayList&lt;String&gt;();
044     * handy.add( list, lines );
045     *
046     * String s = handy.dynReplace("&lt;b&gt;([^&lt;]+)&lt;\\/b&gt;", "&lt;b&gt;b1&lt;/b&gt; txt &lt;b&gt;b2&lt;/b&gt;");
047     * s.equals("b1 txt b2"); // true
048     *
049     * </pre>
050     *
051     * <b>note</b>: There is a lot of comments here since this is
052     * meant to be a library with some documentation, i.e. make 
053     * sure you turn on your comment folding before you start 
054     * reading this file.

065        /**
066         * Join elements of Iterable&lt;? extends Object&gt; with
067         * separator using {@link Object#toString}.
068         *
069         * @param sep the separator to use
070         * @param elements to join into a String
071         * @return all the elements joined by 'sep'
072         */
073        public static String joinAny(String sep,
074                                     Iterable<? extends Object> elements){
075    
076            StringBuilder sb = new StringBuilder("");
077            Iterator<? extends Object> it = elements.iterator();
078    
079            while( it.hasNext() ){
080                sb.append(it.next().toString());
081                if( it.hasNext() ){ sb.append(sep); }
082            }
083    
084            return sb.toString();
085        }

107        /**
108         * Join elements of String array with separator to a String.
109         *
110         * @param sep the separator to use
111         * @param elements to join into a String
112         * @return all the elements joined by 'sep'
113         */
114        public static String join(String sep, String ... elements){
115    
116            StringBuilder sb = new StringBuilder("");
117    
118            for(int i = 0; i < elements.length; i++){
119                sb.append(elements[i]);
120                if(i+1 < elements.length){
121                    sb.append(sep);
122                }
123            }
124            return sb.toString();
125        }

127        /**
128         * Replace every occurence of regexp (containing match-group) with
129         * match-group (sort of like $1 in some regexp implementations).
130         *
131         * @param regexpWithMatchGroup a pattern containing a match-group
132         * @param str the string we want to replace stuff in
133         * @return a string were pattern is replaced with match-group
134         */
135        public static String dynReplace(String regexpWithMatchGroup, String str){
136            return dynReplace( Pattern.compile( regexpWithMatchGroup ), str);
137        }
138    
139        /**
140         * Replace every occurence of pattern (containing match-group) with
141         * match-group (sort of like $1 in some regexp implementations).
142         *
143         * @param patternWithMatchGroup a pattern containing a match-group
144         * @param str the string we want to replace stuff in
145         * @return a string were pattern is replaced with match-group
146         */
147        public static String dynReplace(Pattern patternWithMatchGroup, String str){
148    
149            String tmp = str;
150            Matcher m  = patternWithMatchGroup.matcher(str);
151    
152            while(m.find()){
153                tmp = m.replaceFirst( m.group(1) );
154                m = patternWithMatchGroup.matcher( tmp );
155            }
156    
157            return tmp;
158        }

161        /**
162         * Split URI query into a map.
163         *
164         * NOTE: this version overwrites entries with the same name,
165         * i.e. each key has to be unique for this to work.
166         *
167         * @param uri an URI with a query
168         * @param name of charset for decoding query
169         * @return a map with key value pair from query or null on empty query
170         */
171        public static HashMap<String, String> uriQuery(URI uri, String enc)
172            throws UnsupportedEncodingException
173        {
174            String question = uri.getRawQuery();
175    
176            if(question == null){ return null; }
177    
178            HashMap<String, String> map = new HashMap<String, String>();
179            String[]      keyValuePairs = question.split("&");
180    
181            String key, value;
182            String[] keyValue;
183    
184            for(String kv: keyValuePairs){
185                keyValue = kv.split("=");
186                if( keyValue.length == 2 ){
187                    key   = URLDecoder.decode(keyValue[0], enc);
188                    value = URLDecoder.decode(keyValue[1], enc);
189                    map.put(key, value);
190                }
191            }
192    
193            return map;
194        }

212        /**
213         * Split URI query into a map.
214         *
215         * @param uri an URI with a query
216         * @param name of charset for decoding query
217         * @return a map with key value pairs from query or null on empty query
218         */
219        public static
220            HashMap<String, ArrayList<String>> uriQueryFancy(URI uri, String enc)
221            throws UnsupportedEncodingException
222        {
223            String question = uri.getRawQuery();
224    
225            if(question == null){ return null; }
226    
227            HashMap<String, ArrayList<String>> map = 
228                new HashMap<String, ArrayList<String>>();
229    
230            String[]      keyValuePairs = question.split("&");
231    
232            String key, value;
233            String[] keyValue;
234            ArrayList<String> list;
235    
236            for(String kv: keyValuePairs){
237                keyValue = kv.split("=");
238                if( keyValue.length == 2 ){
239                    key   = URLDecoder.decode(keyValue[0], enc);
240                    value = URLDecoder.decode(keyValue[1], enc);
241                    if( map.containsKey(key) ){
242                        list = map.get( key );
243                        list.add( value );
244                    }else{
245                        list = new ArrayList<String>();
246                        list.add( value );
247                        map.put(key, list);
248                    }
249                }
250            }
251    
252            return map;
253        }

264            return uriQueryFancy(uri, "UTF-8");

267        /**
268         * Utility function to use enumerations in foreach loop.
269         *
270         * @param en Enumeration of some sort
271         * @return a list containing elements from Enumeration
272         */
273        public <T> ArrayList<T> toList(Enumeration<T> en){
274            ArrayList<T> arr = new ArrayList<T>();
275            if( en != null ){
276                while(en.hasMoreElements()){
277                    arr.add( en.nextElement() );
278                }
279            }
280            return arr;
281        }
