059        /**
060         * Compute a security hash of an array of bytes.
061         *
062         * This function is basically just a wrapper for the
063         * standard library {@link MessageDigest} methods.
064         *
065         * @param b bytes to base the hash on
066         * @param algorithm which message digest algorithm to use
067         */
068        public static byte[] digest(byte[] b, String algorithm)
069            throws IOException, NoSuchAlgorithmException
070        {
071            MessageDigest md = MessageDigest.getInstance(algorithm);
072            md.reset();
073            md.update(b);
074            return md.digest();
075        }

240         *
241         * NOTE: this is pretty shitty
242         *
243         * @param  b input bytes
244         * @return the hex representation of the input bytes
245         */
246        public static String toHex(byte[] b){
247            StringBuilder sb = new StringBuilder();
248            for(int i = 0; i < b.length; i++){
249                sb.append(String.format("%02x", b[i]));
250            }
251            return sb.toString();
