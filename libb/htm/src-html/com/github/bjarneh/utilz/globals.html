016    package com.github.bjarneh.utilz;
017    
018    // stdlib
019    import java.util.Hashtable;
020    
021    /**
022     * Global key-value pairs accessible through get/set functions.
023     *
024     * A place to store global values as key value pairs of different
025     * types, only Strings' and some basic types are present, perhaps
026     * an Object Hashtable would be a good idea as well.
027     *
028     * <pre>
029     *
030     * // Typical use:
031     *
032     * globals.set("number", 1);
033     * globals.set("key", "value");
034     *
035     * String key = globals.getStr("key");
036     * int number = globals.getInt("number");
037     *
038     * </pre>
039     *
040     * @version 1.0
041     * @author  bjarneh@ifi.uio.no
042     */
043    
044    public class globals {
045    
046        // One Hashtable for each type, Hashtable is used for thread-safety.
047    
048        // global string values
049        private static Hashtable<String, String> strs = 
050                              new Hashtable<String, String>();
051    
052        // global int values
053        private static Hashtable<String, Integer> ints = 
054                              new Hashtable<String, Integer>();
055    
056        // global long values
057        private static Hashtable<String, Long> longs = 
058                              new Hashtable<String, Long>();
059    
060        // global double values
061        private static Hashtable<String, Double> doubles = 
062                              new Hashtable<String, Double>();
063    
064        // global boolean values
065        private static Hashtable<String, Boolean> bools = 
066                              new Hashtable<String, Boolean>();
067    
068    
069        // we only have functions
070        private globals(){ }
071    
072        /**
073         * Set a global string value.
074         *
075         * @param key name of global string value
076         * @param val value of the global string value
077         */
078        public static void set(String key, String val){ strs.put(key, val); }
079    
080        /**
081         * Set a global int value.
082         *
083         * @param key name of global int value
084         * @param val value of the global int value
085         */
086        public static void set(String key, int val){ ints.put(key, val); }
087    
088        /**
089         * Set a global long value.
090         *
091         * @param key name of global long value
092         * @param val value of the global long value
093         */
094        public static void set(String key, long val){ longs.put(key, val); }
095    
096        /**
097         * Set a global double value.
098         *
099         * @param key name of global double value
100         * @param val value of the global double value
101         */
102        public static void set(String key, double val){ doubles.put(key, val); }
103    
104        /**
105         * Set a global boolean.
106         *
107         * @param key name of global boolean
108         * @param val value of the global boolean
109         */
110        public static void set(String key, boolean val){ bools.put(key, val); }
111    
112        /**
113         * Return global string value
114         *
115         * @param key name of global string value
116         * @return the value of the global variable named key
117         */
118        public static String getStr(String key){ return strs.get(key); }
119    
120        /**
121         * Return global int value
122         *
123         * @param key name of global int value
124         * @return the value of the global variable named key
125         */
126        public static int getInt(String key){ return ints.get(key); }
127    
128        /**
129         * Return global long value
130         *
131         * @param key name of global long value
132         * @return the value of the global variable named key
133         */
134        public static long getLong(String key){ return longs.get(key); }
135    
136        /**
137         * Return global double value
138         *
139         * @param key name of global double value
140         * @return the value of the global variable named key
141         */
142        public static double getDouble(String key){ return doubles.get(key); }
143    
144        /**
145         * Return global boolean variable
146         *
147         * @param key name of global boolean
148         * @return the value of the global variable named key
149         */
150        public static boolean getBoolean(String key){ return bools.get(key); }
151    
152    }
