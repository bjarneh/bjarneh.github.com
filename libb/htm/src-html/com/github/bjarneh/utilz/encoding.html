016    package com.github.bjarneh.utilz;
017    
018    // stdlib
019    import java.io.UnsupportedEncodingException;
020    
021    /**
022     * Convert between different character encodings.
023     *
024     * Utility functions, only a stub for now.
025     *
026     * @version 1.0
027     * @author  bjarneh@ifi.uio.no
028     */
029    
030    public class encoding {
031    
032        // we only have functions
033        private encoding(){}
034    
035        /**
036         * Convert a byte array from one encoding to another.
037         *
038         * @param b bytes array in encoding fromCh
039         * @param fromCh the encoding of the bytes in b
040         * @param toCh the desired encoding of the returned array
041         * @return a byte array in desired encoding
042         */
043        public static byte[] fromTo(byte[] b, String fromCh, String toCh)
044            throws UnsupportedEncodingException
045        {
046            return new String(b, fromCh).getBytes(toCh);
047        }
048    }
