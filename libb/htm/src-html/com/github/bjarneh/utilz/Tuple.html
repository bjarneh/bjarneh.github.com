016    package com.github.bjarneh.utilz;
017    
018    /**
019     * A generic tuple/pair, that is all.
020     *
021     * @version 1.0
022     * @author  bjarneh@ifi.uio.no
023     */
024    
025    public class Tuple<L, R>{
026    
027        private final L left;
028        private final R right;
029    
030        public Tuple(L left, R right){
031            this.left  = left;
032            this.right = right;
033        }
034    
035        public L getLeft(){ return left; }
036        public R getRight(){ return right; }
037    
038        @Override
039        public int hashCode(){
040            return left.hashCode() + 37 * right.hashCode();
041        }
042    
043        @Override
044        public boolean equals(Object other){
045    
046            if(other == this){ return true; }
047    
048            if(other instanceof Tuple){
049    
050                Tuple tupleOther = (Tuple) other;
051    
052                return this.left.equals(tupleOther.getLeft()) &&
053                       this.right.equals(tupleOther.getRight());
054            }
055    
056            return false;
057        }
058    
059        @Override
060        public String toString(){
061            return String.format("Tuple(%s, %s)", getLeft(), getRight());
062        }
063    }
