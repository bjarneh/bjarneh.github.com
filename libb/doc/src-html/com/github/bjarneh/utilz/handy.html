016    package com.github.bjarneh.utilz;
017    
018    // stdlib
019    import java.util.Iterator;
020    
021    /**
022     * Collecting some handy functions that can be useful all over the place.
023     *
024     * Note to self, try to keep this class as independant as possible,
025     * i.e. no dependencies outside the stdlib.
026     *
027     * <b>note</b>: There is a lot of comments here since this is
028     * meant to be a library with some documentation, i.e. make 
029     * sure you turn on your comment folding before you start 
030     * reading this file.
031     *
032     * @version 1.0
033     * @author  bjarneh@ifi.uio.no
034     */
035    
036    public class handy {
037    
038        // we only have functions
039        private handy(){}
040    
041        /**
042         * Join elements of Iterable&lt;? extends Object&gt; with
043         * separator using {@link Object#toString}.
044         *
045         * @param sep the separator to use
046         * @param elements to join into a String
047         * @return all the elements joined by 'sep'
048         */
049        public static String joinAny(String sep,
050                                     Iterable<? extends Object> elements){
051    
052            StringBuilder sb = new StringBuilder("");
053            Iterator<? extends Object> it = elements.iterator();
054    
055            while( it.hasNext() ){
056                sb.append(it.next().toString());
057                if( it.hasNext() ){ sb.append(sep); }
058            }
059    
060            return sb.toString();
061        }
062    
063        /**
064         * Join elements of Iterable&lt;String&gt; with separator to a String.
065         *
066         * @param sep the separator to use
067         * @param elements to join into a String
068         * @return all the elements joined by 'sep'
069         */
070        public static String join(String sep, Iterable<String> elements){
071    
072            StringBuilder sb = new StringBuilder("");
073            Iterator<String> it = elements.iterator();
074    
075            while( it.hasNext() ){
076                sb.append(it.next());
077                if( it.hasNext() ){ sb.append(sep); }
078            }
079    
080            return sb.toString();
081        }
082    
083        /**
084         * Join elements of String array with separator to a String.
085         *
086         * @param sep the separator to use
087         * @param elements to join into a String
088         * @return all the elements joined by 'sep'
089         */
090        public static String join(String sep, String[] elements){
091    
092            StringBuilder sb = new StringBuilder("");
093    
094            for(int i = 0; i < elements.length; i++){
095                sb.append(elements[i]);
096                if(i+1 < elements.length){
097                    sb.append(sep);
098                }
099            }
100            return sb.toString();
101        }
102    
103        /**
104         * Get the hex representation of a byte array.
105         *
106         * NOTE: this is pretty shitty
107         *
108         * @param  b input bytes
109         * @return the hex representation of the input bytes
110         */
111        public static String toHex(byte[] b){
112            StringBuilder sb = new StringBuilder();
113            for(int i = 0; i < b.length; i++){
114                sb.append(String.format("%02x", b[i]));
115            }
116            return sb.toString();
117        }
118    
119    }
