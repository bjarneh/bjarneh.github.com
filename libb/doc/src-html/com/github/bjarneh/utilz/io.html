016package com.github.bjarneh.utilz;
017
018// stdlib
019import java.net.URL;
020import java.io.File;
021import java.io.InputStream;
022import java.io.OutputStream;
023import java.io.FileInputStream;
024import java.io.IOException;
025import java.io.ByteArrayOutputStream;
026import java.nio.MappedByteBuffer;
027import java.nio.channels.FileChannel;
028
029/**
030 * Common io functions.
031 *
032 * <pre>
033 *
034 *  // Typical use:
035 *
036 *  byte[] bytes;
037 *  
038 *  // fetch file as raw byte array
039 *  bytes = io.raw("somefile.txt");
040 *
041 *  // fetch URL as raw byte array
042 *  bytes = io.wget("http://example.com/index.html");
043 *
044 *  // fetch stdin as raw byte array
045 *  bytes = io.raw(System.in);
046 *
047 *  // pipe an input stream to an output stream (stdin to stdout)
048 *  io.pipe(System.in, System.out);
049 *
050 * </pre>
051 * <b>note</b>: There is a lot of comments here since this is
052 * meant to be a library with some documentation, i.e. make 
053 * sure you turn on your comment folding before you start 
054 * reading this file.
055 *
056 * @version 1.0
057 * @author  bjarneh@ifi.uio.no
058 */
059
060public class io {
061
062    // we only have functions
063    private io(){}
064
065    public final static int DEFAULT_BUFFER_SIZE = 4096;
066
067
068    ////////////////////
069    // raw file slurp //
070    ////////////////////
071
072    /**
073     * In memory raw file slurp.
074     *
075     * It should be noted that I'm just guessing that this is the
076     * best way to do things based on votes from stackoverflow,
077     * which could very well be wrong; but if it is I'm not alone :-)
078     *
079     * stackoverflow: 326390
080     *
081     * @param stream to slurp and return as raw bytes
082     * @return a byte array containing the bytes of the stream
083     */
084    public static byte[] raw(FileInputStream stream) throws IOException {
085        try {
086            FileChannel fc = stream.getChannel();
087            MappedByteBuffer bb =
088                fc.map(FileChannel.MapMode.READ_ONLY, 0, fc.size());
089            if(bb.hasArray()){
090                return bb.array();
091            }else{
092                byte[] b = new byte[(int)fc.size()];
093                bb.get(b);
094                return b;
095            }
096        }
097        finally {
098            stream.close();
099        }
100    }
101
102    /**
103     * 
104     * Alias for io.raw(new File(fname)).
105     *
106     * @param fname file-name to slurp and return as raw bytes
107     * @return a byte array containing the bytes of the file 
108     */
109    public static byte[] raw(String fname) throws IOException {
110        return raw(new File(fname));
111    }
112
113    /**
114     * Alias for io.raw(new FileInputStream(file)).
115     *
116     * @param file file object to slurp and return as raw bytes
117     * @return a byte array containing the bytes of the file 
118     */
119    public static byte[] raw(File file) throws IOException {
120        return raw(new FileInputStream(file));
121    }
122
123    /**
124     * Use io.pipe to write to an ByteArrayOutputStream, and fetch array.
125     *
126     * @param input the stream we want to return as raw bytes
127     * @return a byte array containing the bytes of the input stream
128     */
129    public static byte[] raw(InputStream input) throws IOException {
130        ByteArrayOutputStream output = new ByteArrayOutputStream();
131        pipe(input, output);
132        return output.toByteArray();
133    }
134
135
136    ////////////////////////////////////////////
137    // piping input streams to output streams //
138    ////////////////////////////////////////////
139
140    /**
141     * Write bytes from an input stream to an output stream.
142     *
143     * @param fi  where bytes are read
144     * @param fo  where bytes are written
145     * @param bufferSize is the size of the buffer used
146     */
147    public static void pipe(InputStream fi, OutputStream fo, int bufferSize)
148        throws IOException
149    {
150        int got = -1;
151        byte[] b = new byte[bufferSize];
152
153        for( got = fi.read(b); got > 0; got = fi.read(b) ){
154            fo.write(b, 0, got);
155        }
156    }
157
158    /**
159     * Alias for io.pipe(fi, fo, io.DEFAULT_BUFFER_SIZE).
160     *
161     * @param fi  where bytes are read
162     * @param fo  where bytes are written
163     */
164    public static void pipe(InputStream fi, OutputStream fo)
165        throws IOException
166    {
167        pipe(fi, fo, DEFAULT_BUFFER_SIZE);
168    }
169
170    /**
171     * Fetch the bytes of an String representing an URL.
172     * @param url represented as a String
173     * @return an array of bytes fetched from URL
174     */
175    public static byte[] wget(String url)
176        throws IOException
177    {
178        return wget(new URL(url));
179    }
180
181    /**
182     * Fetch the bytes of an URL.
183     * @param url to read
184     * @return an array of bytes fetched from URL
185     */
186    public static byte[] wget(URL url)
187        throws IOException
188    {
189        return raw(url.openStream());
190    }
191
192    /**
193     * Write the bytes of an URL to an OutputStream.
194     * @param url to read from
195     * @param fo output stream to write bytes to
196     * @param bufferSize size of buffer used by io.pipe
197     */
198    public static void wget(URL url, OutputStream fo, int bufferSize)
199        throws IOException
200    {
201        pipe(url.openStream(), fo, bufferSize);
202    }
203
204    /**
205     * Alias for wget(url, fo, io.DEFAULT_BUFFER_SIZE).
206     * @param url to read from
207     * @param fo output stream to write bytes to
208     */
209    public static void wget(URL url, OutputStream fo)
210        throws IOException
211    {
212        pipe(url.openStream(), fo, DEFAULT_BUFFER_SIZE);
213    }
214}
