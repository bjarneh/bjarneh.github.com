016package com.github.bjarneh.utilz;

018// stdlib
019import java.io.File;
020import java.io.FileFilter;
021import java.util.LinkedList;
022
023// local
024import com.github.bjarneh.utilz.handy;

026/**
027 * Some common path utility functions.
028 *
029 * <pre>
030 *
031 *  // Typical use:
032 *
033 *  String validPathOnYourOs;
034 *
035 *  validPathOnYourOs = path.fromSlash("/some/file/name");
036 *  validPathOnYourOs = path.join(new String[]{"some","dir","name"});
037 *
038 *  // get a list of files under 'validPathOnYourOs'
039 *  String[] files    = path.walk(validPathOnYourOs);
040 * </pre>
041 *
042 * <b>note</b>: There is a lot of comments here since this is
043 * meant to be a library with some documentation, i.e. make 
044 * sure you turn on your comment folding before you start 
045 * reading this file.
046 *
047 * @version 1.0
048 * @author  bjarneh@ifi.uio.no
049 */
050
051public class path {
052
053    // we only have functions
054    private path(){}
055
056    private static final String OS_SEP = File.separator;
057
058    // filter that does zero filtering says yes to anything
059    private static final FileFilter yesFilter = new FileFilter() {
060        @Override public boolean accept(File file){
061            return true;
062        }
063    };
064
065    // filter that ignores common vcs directories (.bzr/.git/.hg/.cvs/.svn)
066    private static final FileFilter vcsFilter = new FileFilter() {
067        @Override public boolean accept(File file){
068            if( file.isDirectory() ){
069                String fname = file.getName();
070                if(fname.equals(".git")||
071                   fname.equals(".svn")||
072                   fname.equals(".hg") ||
073                   fname.equals(".bzr")||
074                   fname.equals(".cvs"))
075                {
076                    return false;
077                }
078            }
079            return true;
080        }
081    };
082
083
084    ///////////////
085    // path walk //
086    ///////////////
087
088
089    /**
090     * Fetch the root relative path names of all/some files under root.
091     *
092     * @param root of the file tree you would like to fetch
093     * @param filter out parts of your file tree using a FileFilter
094     * @return a list of path names under root
095     */
096    public static String[] walk(String root, FileFilter filter){
097
098        File fileRoot = new File(root);
099        if( ! fileRoot.isDirectory() ){ return new String[0]; }
100
101        LinkedList<String> list = new LinkedList<String>();
102        walk(clean(root), fileRoot, list, filter);
103
104        String[] arr = new String[list.size()];
105        list.toArray(arr);
106
107        return arr;
108    }
109
110    /**
111     * Fetch the root relative path names of all files under root.
112     *
113     * @param root of the file tree you would like to fetch
114     * @return a list of path names under root
115     */
116    public static String[] walk(String root){
117        return walk(clean(root), yesFilter);
118    }
119
120    /**
121     * Fetch the root relative path names of all files under root,
122     * and ignore VCS dirs (.bzr/.git/.hg/.svn/.cvs).
123     *
124     * @param root of the file tree you would like to fetch
125     * @return a list of path names under root, ignoring VCS dirs
126     */
127    public static String[] walkVCS(String root){
128        return walk(clean(root), vcsFilter);
129    }
130
131
132    private static void walk(String root,
133                             File dir,
134                             LinkedList<String> save,
135                             FileFilter filter)
136    {
137        File[] ls = dir.listFiles(filter);
138
139        for(File sub : ls){
140            if(sub.isFile()){
141                save.add(root + OS_SEP + sub.getName());
142            }else if(sub.isDirectory()){
143                walk(root + OS_SEP + sub.getName(),
144                     new File(dir, sub.getName()), save, filter);
145            }
146        }
147    }
148
149    /**
150     * This function cleans up a path name, by removing
151     * duplicates and ending File.separator.
152     *
153     * @param p a path name
154     * @return a path with no ending or duplicate File.separator
155     */
156    public static String clean(String p){
157
158        while(p.length() > 1 && p.endsWith(OS_SEP)){
159            p = p.substring(0, p.length() - 1);
160        }
161
162        StringBuilder sb = new StringBuilder();
163        char[] pChars = p.toCharArray();
164        char prev = ' ';
165        char sep  = File.separatorChar;
166
167        for(int i = 0; i < pChars.length; i++){
168            if(pChars[i] == sep && prev == sep){
169                continue;
170            }
171            sb.append(pChars[i]);
172            prev = pChars[i];
173        }
174
175        return sb.toString();
176    }
177
178    /**
179     * Replace forward slash with File.separator.
180     *
181     * @param p name you would like to OS-ify
182     * @return a path name that fits your OS
183     */
184    public static String fromSlash(String p){
185        return p.replaceAll("/", OS_SEP);
186    }
187
188    /**
189     * Join with File.separator as separator.
190     * @param elements to join into a String
191     * @return a File.separator joined String
192     */
193    public static String join(Iterable<String> elements){
194        return handy.join(OS_SEP, elements);
195    }
196
197    /**
198     * Join with File.separator as separator.
199     * @param elements to join into a String
200     * @return a File.separator joined String
201     */
202    public static String join(String[] elements){
203        return handy.join(OS_SEP, elements);
204    }
