016package com.github.bjarneh.utilz;
017
018// stdlib
019import java.util.Hashtable;
020
021/**
022 * Global key-value pairs accessible through get/set functions.
023 *
024 * A place to store global values as key value pairs of different
025 * types, only Strings' and some basic types are present, perhaps
026 * an Object Hashtable would be a good idea as well.
027 *
028 * <pre>
029 *
030 * // Typical use:
031 *
032 * globals.set("number", 1);
033 * globals.set("key", "value");
034 *
035 * // fetch them at some point
036 *
037 * String key = globals.getStr("key");
038 * int number = globals.getInt("number");
039 *
040 * </pre>
041 *
042 * @version 1.0
043 * @author  bjarneh@ifi.uio.no
044 */
045
046public class globals {
047
048    // One Hashtable for each type, Hashtable is used for thread-safety.
049
050    // global string values
051    private static Hashtable<String, String> strs = 
052                          new Hashtable<String, String>();
053
054    // global int values
055    private static Hashtable<String, Integer> ints = 
056                          new Hashtable<String, Integer>();
057
058    // global long values
059    private static Hashtable<String, Long> longs = 
060                          new Hashtable<String, Long>();
061
062    // global double values
063    private static Hashtable<String, Double> doubles = 
064                          new Hashtable<String, Double>();
065
066    // global boolean values
067    private static Hashtable<String, Boolean> bools = 
068                          new Hashtable<String, Boolean>();
069
070
071    // we only have functions
072    private globals(){ }
073
074    /**
075     * Set a global string value.
076     *
077     * @param key name of global string value
078     * @param val value of the global string value
079     */
080    public static void set(String key, String val){ strs.put(key, val); }
081
082    /**
083     * Set a global int value.
084     *
085     * @param key name of global int value
086     * @param val value of the global int value
087     */
088    public static void set(String key, int val){ ints.put(key, val); }
089
090    /**
091     * Set a global long value.
092     *
093     * @param key name of global long value
094     * @param val value of the global long value
095     */
096    public static void set(String key, long val){ longs.put(key, val); }
097
098    /**
099     * Set a global double value.
100     *
101     * @param key name of global double value
102     * @param val value of the global double value
103     */
104    public static void set(String key, double val){ doubles.put(key, val); }
105
106    /**
107     * Set a global boolean.
108     *
109     * @param key name of global boolean
110     * @param val value of the global boolean
111     */
112    public static void set(String key, boolean val){ bools.put(key, val); }
113
114    /**
115     * Return global string value
116     *
117     * @param key name of global string value
118     * @return the value of the global variable named key
119     */
120    public static String getStr(String key){ return strs.get(key); }
121
122    /**
123     * Return global int value
124     *
125     * @param key name of global int value
126     * @return the value of the global variable named key
127     */
128    public static int getInt(String key){ return ints.get(key); }
129
130    /**
131     * Return global long value
132     *
133     * @param key name of global long value
134     * @return the value of the global variable named key
135     */
136    public static long getLong(String key){ return longs.get(key); }
137
138    /**
139     * Return global double value
140     *
141     * @param key name of global double value
142     * @return the value of the global variable named key
143     */
144    public static double getDouble(String key){ return doubles.get(key); }
145
146    /**
147     * Return global boolean variable
148     *
149     * @param key name of global boolean
150     * @return the value of the global variable named key
151     */
152    public static boolean getBoolean(String key){ return bools.get(key); }
153
154}
